                            function locationAutocomplete() {
                                var autocomplete = new google.maps.places.Autocomplete(document.getElementById('address'), {
                                    componentRestrictions: {country: "au"},
                                    types: ['address']
                                });
                                autocomplete.addListener('place_changed', function () {
                                    var place = autocomplete.getPlace();

                                    app.form.zip = '';
                                    app.form.street_number = '';
                                    app.form.route = '';
                                    app.form.city = '';
                                    app.form.state = '';
                                    app.form.state_full = 'your state';
                                    app.streeterror=false;

                                    if (typeof place.address_components != 'undefined') {
                                        for (i = 0; i < place.address_components.length; i++) {
                                            if (place.address_components[i].types.indexOf('postal_code') != -1) {
                                                app.form.zip = place.address_components[i].long_name;

                                            }
                                            if (place.address_components[i].types.indexOf('street_number') != -1) {
                                                app.form.street_number = place.address_components[i].long_name;
                                            }
                                            if (place.address_components[i].types.indexOf('route') != -1) {
                                                app.form.route = place.address_components[i].long_name;
                                            }
                                            if (place.address_components[i].types.indexOf('locality') != -1) {
                                                app.form.city = place.address_components[i].long_name;
                                            }
                                            if (place.address_components[i].types.indexOf("administrative_area_level_1") != -1) {
                                                app.form.state = place.address_components[i].short_name;
                                                app.form.state_full = place.address_components[i].long_name;
                                            }

                                        }
                                        app.addresserror=false;
                                        app.selectedAddress=true;
                                        if (!app.form.zip) {
                                            app.addresserror=true;
                                        }
                                        if (!app.form.route && !app.form.street_number) {
                                            app.streeterror=true;
                                        }else{
                                            app.streeterror=false;
                                        }
                                        // app.streeterror=false;
                                        if (app.form.street_number && app.form.route) {
//                    $('#address').val(street_number + ' ' + route);
                                        }
                                        else {
//                    $('#address').val(place.formatted_address);
                                        }
                                        app.form.address = $("#address").val();

                                        app.form.latitude = place.geometry.location.lat();
                                        app.form.longitude = place.geometry.location.lng();
                                    }
                                    if (!app.form.zip) {
                                        app.addresserror=true;
                                    }


                                });
                            }
